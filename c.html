    b = (double *)malloc(n * sizeof(double));

        scanf("%lf", &a[i]);
    printf("Please input c:\n");

//void multiply(double a[], int n, double c, double b[])
void multiply(double *a, int n, double c, double *b)
{
    for (int i = 0; i < n; i ++)
        b[i] = c * a[i];
}

#include <stdio.h>
#include <stdlib.h>
int main()
{
    int m, n;
    int *d;
    FILE *fp = fopen("matrix1.txt", "r");
    fscanf(fp, "%d", &m);
    fscanf(fp, "%d", &n);
    d = (int *)malloc(m * n * sizeof(int));
    for (int i = 0; i < m; i ++)
        for (int j = 0; j < n; j ++)
            fscanf(fp, "%d", d + i * m + j);
    printf("%d ", m);
    printf("%d\n", n);
    for (int i = 0; i < m; i ++) {
        for (int j = 0; j < n; j ++)
            printf("%d ", *(d + i *m + j));
        printf("\n");
    }
    return 0;
}

#include <stdio.h>
#include <stdlib.h>
int main()
{
    int m, n;
    int *d;
    FILE *in = fopen("matrix1.txt", "r");
    fscanf(in, "%d", &m);
    fscanf(in, "%d", &n);
    d = (int *)malloc(m * n * sizeof(int));
    for (int i = 0; i < m; i ++)
        for (int j = 0; j < n; j ++)
            fscanf(in, "%d", d + i * m + j);
    FILE *out = fopen("matrix1.dat", "wb");
    fwrite(&m,sizeof(int), 1, out);
    fwrite(&n,sizeof(int), 1, out);
    fwrite(d, sizeof(int), m * n, out);
    fclose(out);
    return 0;
}
