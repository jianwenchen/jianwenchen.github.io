{
    for (int i = 0; i < n; i ++)

}

/*
 *从键盘接收任意多个整数，在屏幕上输出。
 */
#include <stdio.h>
#include <stdlib.h>
struct List {
    int *elem;
    int size; // number of elements;
    int capacity; //size of allocated storage capacity
};
int main()
{
    struct List v1;
    v1.elem = NULL;
    v1.size = 0;
    v1.capacity = 0;
    int x;
    while (1) {
        int c;
        while((c = getchar()) == ' ');
        if (c == '\n')
            break;
        else
            ungetc(c, stdin);
        if (scanf("%d", &x) != 1) break;
        if (v1.size == v1.capacity) {
            int newalloc = (v1.capacity + 1) * 2;
            int *p = (int *)malloc( newalloc * sizeof(int));
            v1.capacity = newalloc;
            for (int i = 0; i < v1.size; i ++)
                p[i] = v1.elem[i];
            free(v1.elem);
            v1.elem = p;
        }
        v1.elem[v1.size++] = x;
    }
    for (int i = 0; i < v1.size; i ++)
        printf("%d ", v1.elem[i]);
    free(v1.elem);
    return 0;
}

/*
 *数据类型=数据+操作
 *数据用变量存放，变量的类型可以是int,float,double,char, char *, array, struct, pointer
 *操作用函数实现
 *本代码为用顺序存储结构实现线性表[a0,a1,...,an-1]的第一个版本
 */
#include <stdio.h>
#include <stdlib.h>
struct List {
    int *elem;
    int size; // number of elements;
    int capacity; //size of allocated storage capacity
};
void List_init(struct List *p);
void push_back(struct List *p, int val);
void List_input(struct List *p);
void List_output(struct List *p);
void List_clear(struct List *p);
int at(struct List *p, int i);
int main()
{
    struct List v1;
    List_init(&v1);
    List_input(&v1);
    struct List v2;
    List_init(&v2);
    List_input(&v2);
    struct List v3;
    List_init(&v3);
    for (int i = 0; i < v1.size; i ++) {
        push_back(&v3, at(&v1, i) + at(&v2, i));
    }
    List_output(&v3);
    List_clear(&v1);
    List_clear(&v2);
    List_clear(&v3);
    return 0;
}
void List_init(struct List *p)
{
    p->elem = NULL;
    p->size = 0;
    p->capacity = 0;
}
void push_back(struct List *p, int val)
{
    if (p->size == p->capacity)
    {
        int newalloc = (p->capacity + 1) * 2;
        int *q = (int *)malloc(newalloc * sizeof(int));
        p->capacity = newalloc;
        for (int i = 0; i < p->size; i++)
            q[i] = p->elem[i];
        free(p->elem);
        p->elem = q;
    }
    p->elem[p->size++] = val;
}
void List_input(struct List *p)
{
    int x;
    while (1) {
        int c;
        while((c = getchar()) == ' ');
        if (c == '\n')
            break;
        else
            ungetc(c, stdin);
        if (scanf("%d", &x) != 1) break;
        push_back(p, x);
    }
}

void List_output(struct List *p)
{
    for (int i = 0; i < p->size; i ++)
        printf("%d ", p->elem[i]);
}
int at(struct List *p, int i)
{
    return p->elem[i];
}
void List_clear(struct List *p)
{
    free(p->elem);
}
